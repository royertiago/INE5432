\begin{frame}
    \frametitle{Definição do predicado}

    Seja $k$ um inteiro positivo,\\
    $a$ um parâmetro,\\
    e $E$ um conjunto.

    \begin{equation*}
        \sigma_{k, a}(E)
    \end{equation*}

    são os $k$ elementos de $E$ mais próximos de $a$.
